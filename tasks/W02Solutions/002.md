     class Solution {
      public:
        bool isPalindrome(int x) {
        if (x < 0) {
            return false;
        }

        int reversed = 0;

        for (int i = x; i >= 1; i/=10)
        {
            reversed *= 10;
            reversed += i % 10;
        }

        return reversed == x;
    }
};
